16  package com.facebook.collections;
17  
18  /**
19   * The idea here is to provide a uniform interface for arrays so that we may
20   * implement data structures on top of them seeming to use "Objects", with auto[un]boxing,
21   * but the backing storage may be primitive arrays in order to save memory
22   * (ex: hash tables implemented in terms of arrays)
23   *
24   * While this interface doesn't 100% enforce that implementations will be a class array,
25   * the implementation should have array-like properties in terms of memory usage and time
26   * of operations:  get/setshould be O(1).  resize should be O(n).  append() should be O(1)
27   * amortized with set() calls in that it's basically an iterator walking towards the end, and will
28   * insert values in the first empty slot. A slot at it's ptr is not empty only if set was called,
29   * so you can see the # of total set/append calls to fill a list is O(N), 2N in the pathological
30   * case (call set on all values and leave the last empty; append will take O(N) time and then
31   * the array is full
32   * 
33   * @param <T> type of the array. T = Long may in fact mean the backing storage
34   * could be long[] which saves 24 bytes per entry (the pointer + 2w overhead)
35   *
36   * Each implementation defines if it's thread safe or not
37   */
38  public interface Array<T> extends Iterable<T>{
39    /**
40     * gets element at position i.  For the love of all that is good and
41     * holy, please make this O(1) in all your implementations
42     * 
43     * @param i position 
44     * @return value at position i, null if empty
45     * @throws ArrayIndexOutOfBoundsException if array is too small, or i is negative
46     */
47    public T get(int i) throws IndexOutOfBoundsException;
48  
49    /**
50     * return size of array (# of elements)
51     * 
52     * @return
53     */
54    public int size();
55  
56    /**
57     * return array capacity (size + empty slots)
58     * same as array.length in the case of built-in arrays
59     *
60     * @return
61     */
62    public int capacity();
63  
64    /**
65     * replace a value.  i must be < capacity()
66     *
67     * @param i location
68     * @param value value to place, may NOT be null
69     * @return previous value, null if nada
70     * @throws IndexOutOfBoundsException i is >= size or i is negative
71     */
72    public T set(int i, T value) throws IndexOutOfBoundsException;
73  
74    /**
75     * add an element to the end of the array. Almost always shorthand for:
76     * 
77     * <pre>simpleArray.set(simpleArray.size(), value) </pre>
78     * (with an exception, see below)
79     *
80     * this is useful when you want to use the array like queue or stack, and in practice some
81     * one-off impls of classes like this did "append" only.
82     *
83     * with the difference that if size() == capacity(), the array it is automatically grown
84     * to accommodate the new value
85     *
86     * In the case if a value has been set past the "end" of contiguous values, this will continue
87     * and skip over set values and set the first empty value
88     * 
89     * @param value to add, may NOT be null
90     * @return position inserted at
91     */
92  
93    public int append(T value);
94  
95    /**
96     * remove a value at position i
97     *
98     * @param i position between 0 and size-
99     * @return value at the position, null if it was empty
100    * @throws ArrayIndexOutOfBoundsException i is >= size or i is negative
101    */
102   public T remove(int i) throws ArrayIndexOutOfBoundsException;
103 
104 
105   /**
106    * the idea here is that want to grow the array, and give a 'hint'. Note,
107    * underlying implementations may not be able to create an array of the exact size
108    * For example, asking for 2 * size() might result in a slightly different size.
109    *
110    * For shrinking: one should copy to a smaller array. Because shrinking would have unknown
111    * effects on the indices of values known to the caller, there is not much value in having
112    * the ability to shrink internalized
113    *
114    * @param sizeHint can be thought of as a new size to grow to, but implementations
115    *                 may not grow to exactly that size. The actual size is returned. Should be
116    *                 > 0
117    * @return actual size of new array
118    */
119   public int resize(int sizeHint);
120 }
