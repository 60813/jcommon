16  package com.facebook.concurrency;
17  
18  import com.facebook.util.exceptions.ExceptionHandler;
19  
20  import java.util.concurrent.Callable;
21  
22  public class CallableSnapshotFunctionImpl<I, O, E extends Exception> implements
23    CallableSnapshotFunction<I,O,E> {
24    private final ValueFactory<I, O, E> valueFactory;
25    private final ExceptionHandler<E> exceptionHandler;
26  
27    public CallableSnapshotFunctionImpl(
28      ValueFactory<I, O, E> valueFactory, ExceptionHandler<E> exceptionHandler
29    ) {
30      this.valueFactory = valueFactory;
31      this.exceptionHandler = exceptionHandler;
32    }
33  
34    public CallableSnapshotFunctionImpl(ValueFactory<I, O, E> valueFactory) {
35      // We can cast exceptions because the value factory declares which type
36      // of exceptions it can throw on creation
37      this(valueFactory, new CastingExceptionHandler<E>());
38    }
39  
40    @Override
41    public CallableSnapshot<O, E> apply(final I input) {
42      return new CallableSnapshot<O, E>(
43        new Callable<O>() {
44          @Override
45          public O call() throws E {
46            return valueFactory.create(input);
47          }
48        },
49        exceptionHandler
50      );
51    }
52  }
