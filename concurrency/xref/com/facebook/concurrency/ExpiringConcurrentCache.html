104     long maxAge,

106     ExceptionHandler<E> exceptionHandler,

108   ) {

110       valueFactory,

114         @Override
115         public void evicted(K key, V value) {
116           try {

179   /**
180    * non-blocking, thread-safe prune operation that only enters pruning block
181    * after enough time has elapsed
182    *
183    * @throws E
184    */
185   private void pruneIfNeeded() {
186     // only prune if sufficient time has elapsed and another thread isn't
187     // already pruning
188     if (DateTimeUtils.currentTimeMillis() - lastPrune.get() >= maxAgeMillis &&
189       pruning.compareAndSet(false, true)) {
190       try {
191         Iterator<Map.Entry<K, CallableSnapshot<CacheEntry<V, E>, E>>> iterator =
192           baseCache.iterator();
193 
194         while (iterator.hasNext()) {
195           final K key;
196           final CacheEntry<V, E> cacheEntry;
197           try {
198             Map.Entry<K, CallableSnapshot<CacheEntry<V, E>, E>> entry =
199               iterator.next();
200             key = entry.getKey();
201             cacheEntry = entry.getValue().get();
202           } catch (Exception e) {
203             // We control the creation process, so should not get an exception
204             throw new RuntimeException("CacheEntry create should not fail");
205           }
206 
207           if (cacheEntry.hasExpired(maxAgeMillis)) {
208             // remove the item from the cache
209             iterator.remove();
210 
211             // do any shutdown() tasks asynchronously so we don't block access
212             // to the cache
213             executor.execute(
214               new Runnable() {
215                 @Override
216                 public void run() {
217                   // now reap the entry
218                   try {
219                     V value = cacheEntry.getSnapshot().get();
220 
221                     try {
222                       evictionListener.evicted(key, value);
223                     } catch (Throwable t) {
224                       LOG.error(
225                         "Error reaping cache element-- may not be properly closed",
226                         t
227                       );
228                     }
229                   } catch (Exception e) {
230                     LOG.info(
231                       "Unable to get cache value for key " + key
232                     );
233                     // still notify that key is evicted
234                     evictionListener.evicted(key, null);
235                   }
236                 }
237               }
238             );
239 
240           }
241         }
242       } finally {
243         lastPrune.set(DateTimeUtils.currentTimeMillis());
244         pruning.set(false);
245       }
246     }
247   }

259   @Override
260   public CallableSnapshot<V, E> getIfPresent(K key) {
261     pruneIfNeeded();
262     CallableSnapshot<CacheEntry<V, E>, E> snapshot =
263       baseCache.getIfPresent(key);
264 
265     if (snapshot == null) {
266       return null;
267     } else {
268       try {
269         return snapshot.get().getSnapshot();
270       } catch (Exception e) {
271         throw new RuntimeException("this shouldn't happen", e);
272       }
273     }

276   private class ValueMapper implements
277     Mapper<
278       Map.Entry<K, CallableSnapshot<CacheEntry<V, E>, E>>,
279       Map.Entry<K, CallableSnapshot<V, E>>
280       > {
281     @Override
282     public Map.Entry<K, CallableSnapshot<V, E>> map(
283       Map.Entry<K, CallableSnapshot<CacheEntry<V, E>, E>> input
284     ) {
285       CallableSnapshot<V, E> snapshot;
286       try {
287         snapshot = input.getValue().get().touch();
288       } catch (Exception e) {
289         // We control the creation process, so we should not get an exception
290         throw new RuntimeException("CacheEntry create should not fail");
291       }
292       return new AbstractMap.SimpleImmutableEntry<K, CallableSnapshot<V, E>>(
293         input.getKey(),
294         snapshot
295       );
296     }

314   /**
315    * a cache entry is a value and it's last accessed time (create, read).
316    * The last accessed is used for expiring entire older than a configured
317    * TTL by the cache
318    *
319    * @param <V> value type
320    * @param <E> exception type
321    */
322   @SuppressWarnings({"unchecked"})
323   private static class CacheEntry<V, E extends Exception> {
324     // mtime guarded by this
325     private long mtime = DateTimeUtils.currentTimeMillis();
326     private volatile Object snapshotOrValue;
327 
328     private CacheEntry(V value) {
329       this.snapshotOrValue = value;
330     }
331 
332     private CacheEntry(CallableSnapshot<V, E> snapshot) {
333       // if the snapshot indicates no error, store just the value. This
334       // will save us about 24 bytes on a 64-bit box:  2 x 8 byte ptr and
335       // the 8-byte overhead java adds for each object (saved by not
336       // having the CallableSnapshot)
337       if (snapshot.getException() == null) {
338         try {
339           snapshotOrValue = snapshot.get();
340         } catch (Exception e) {
341           LOG.error("this should NEVER be seen", e);
342           snapshotOrValue = snapshot;
343         }
344       } else {
345         snapshotOrValue = snapshot;
346       }
347     }
348 
349     public CallableSnapshot<V, E> getSnapshot() throws E {
350       return getCallableSnapshot();
351     }
352 
353     public synchronized CallableSnapshot<V, E> touch() {
354       mtime = DateTimeUtils.currentTimeMillis();
355 
356       return getCallableSnapshot();
357     }
358 
359     public synchronized boolean hasExpired(long maxAgeMillis) {
360       return DateTimeUtils.currentTimeMillis() - mtime >= maxAgeMillis;
361     }
362 
363     /**
364      * we store either the result of the Callable if there is no 
365      * exception (saves memory). Otherewise, we keep the whole CallableSnapshot
366      *
367      * @return
368      */
369     private CallableSnapshot<V, E> getCallableSnapshot() {
370       if (snapshotOrValue instanceof PrivateCallableSnapshot) {
371         return (CallableSnapshot<V, E>) snapshotOrValue;
372       } else {
373         // we can use NullExceptionHandler since we know
374         // this is a value
375         return new PrivateCallableSnapshot<V, E>(
376           new FixedValueCallable<V>((V) snapshotOrValue),
377           new NullExceptionHandler<E>()
378         );
379       }
380     }
381   }

383   // this private class is using it's class type as a boolean flag (to save
384   // memory). If the value stored is of this type in the CacheEntry,
385   // then it means we couldn't store the value and need to call 
386   // CallableSnapshot.get(). By making it private, we guarantee that O
387   // cannot be this type
388   private static class PrivateCallableSnapshotFunction
389     <I, O, E extends Exception>
390     implements CallableSnapshotFunction<I, O, E> {
391 
392     private final ValueFactory<I, O, E> valueFactory;
393     private final ExceptionHandler<E> exceptionHandler;
394 
395     private PrivateCallableSnapshotFunction(
396       ValueFactory<I, O, E> valueFactory, ExceptionHandler<E> exceptionHandler
397     ) {
398       this.valueFactory = valueFactory;
399       this.exceptionHandler = exceptionHandler;
400     }
401 
402     private PrivateCallableSnapshotFunction(ValueFactory<I, O, E> valueFactory) {
403       // We can cast exceptions because the value factory declares which type
404       // of exceptions it can throw on creation
405       this(valueFactory, new CastingExceptionHandler<E>());
406     }
407 
408     @Override
409     public CallableSnapshot<O, E> apply(final I input) {
410       return new PrivateCallableSnapshot<O, E>(
411         new Callable<O>() {
412           @Override
413           public O call() throws E {
414             return valueFactory.create(input);
415           }
416         },
417         exceptionHandler
418       );
419     }
420 
421   }
