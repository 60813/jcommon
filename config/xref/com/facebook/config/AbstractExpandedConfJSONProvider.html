29  /**
30   * Reads and expands a JSON config object through a series of includes.
31   * Expects JSON config objects to consist of two top level keys: 'conf' and 'includes':
32   *
33   * conf - key-value pairs to be consolidated in the returned JSONObject [req]
34   * includes - list of JSON configs (of the same format) from which to pull and
35   *    include key-value pairs [opt]
36   *
37   * JSON config file format:
38   * {
39   *  conf : {
40   *    key1 : value1,
41   *    key2 : value2
42   *    ...
43   *  },
44   *  includes : [
45   *    object1,
46   *    object2
47   *   ]
48   * }
49   *
50   * Keys in the existing or closer config objects will take precedence over the same
51   * key defined in more distantly included objects.
52   */
53  public abstract class AbstractExpandedConfJSONProvider implements JSONProvider {
54    private static final Logger LOG = Logger.getLogger(AbstractExpandedConfJSONProvider.class);
55    private static final String CONF_KEY = "conf";
56    private static final String INCLUDES_KEY = "includes";
57  
58    private final String root;
59  
60    public AbstractExpandedConfJSONProvider(String root) {
61      this.root = root;
62    }
63  
64    private JSONObject getExpandedJSONConfig() throws JSONException {
65      Set<String> traversedFiles = new HashSet<>();
66      Queue<String> toTraverse = new LinkedList<>();
67  
68      // Seed the graph traversal with the root node
69      traversedFiles.add(root);
70      toTraverse.add(root);
71  
72      // Policy: parent configs will override children (included) configs
73      JSONObject expanded = new JSONObject();
74      while (!toTraverse.isEmpty()) {
75        String current = toTraverse.remove();
76        JSONObject json = load(current);
77        JSONObject conf = json.getJSONObject(CONF_KEY);
78        Iterator<String> iter = conf.keys();
79        while (iter.hasNext()) {
80          String key = iter.next();
81          // Current config will get to insert keys before its include files
82          if (!expanded.has(key)) {
83            expanded.put(key, conf.get(key));
84          }
85        }
86        // Check if the file itself has any included files
87        if (json.has(INCLUDES_KEY)) {
88          JSONArray includes = json.getJSONArray(INCLUDES_KEY);
89          for (int idx = 0; idx < includes.length(); idx++) {
90            String include = resolve(current, includes.getString(idx));
91            if (traversedFiles.contains(include)) {
92              LOG.warn("Config file was included twice: " + include);
93            } else {
94              toTraverse.add(include);
95              traversedFiles.add(include);
96            }
97          }
98        }
99      }
100 
101     return expanded;
102   }
103 
104   @Override
105   public JSONObject get() throws JSONException {
106     return getExpandedJSONConfig();
107   }

109   /**
110    * Determines the canonical resource identifier for the given <tt>config</tt>.
111    * This hook allows subclasses to resolve relative paths used in includes.
112    *
113    * @param parent the resource which listed this config in its includes
114    *               (or <tt>null</tt> if it is the root)
115    * @param config a config resource identifier
116    * @return the canonical resource identifier
117    * @see com.facebook.config.ExpandedConfFileJSONProvider#resolve(String, String)
118    */
119   abstract protected String resolve(String parent, String config);

121   /**
122    * Returns the configuration JSON identified by <tt>config</tt>.
123    * <tt>config</tt> is guaranteed to be the result of some call to
124    * {@link #resolve(String, String)}.
125    *
126    * @param config a canonical resource identifier
127    * @return a JSON representation of the resource's contents
128    * @see com.facebook.config.ExpandedConfFileJSONProvider#load(String)
129    */
130   abstract protected JSONObject load(String config) throws JSONException;
