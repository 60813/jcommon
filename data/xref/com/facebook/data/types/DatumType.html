25  /**
26   * general type of a Datum; this can help a caller determine the type
27   * of the asRaw() method and is used for serializing Datums
28   */
29  public enum DatumType {
30    NULL(new NullDatum.SerDeImpl(), -1),
31    BOOLEAN(new BooleanDatum.SerDeImpl(), 0),
32    BYTE(new ByteDatum.SerDeImpl(), 1),
33    CHAR(new ByteDatum.SerDeImpl(), 2), // TOOD: not used yet
34    SHORT(new ShortDatum.SerDeImpl(), 3),
35    INTEGER(new IntegerDatum.SerDeImpl(), 4),
36    LONG(new LongDatum.SerDeImpl(), 5),
37    FLOAT(new FloatDatum.SerDeImpl(), 6),
38    DOUBLE(new DoubleDatum.SerDeImpl(), 7),
39    STRING(new StringDatum.SerDeImpl(), 8),
40    LIST(new ListDatum.SerDeImpl(), 9),
41    MAP(new MapDatum.SerDeImpl(), 10),
42    OTHER(null, 127), // TODO ?
43    ;
44  
45    private final SerDe<Datum> serDe;
46    private final byte typeAsByte;
47  
48    DatumType(SerDe<Datum> serDe, byte typeAsByte) {
49      this.serDe = serDe;
50      this.typeAsByte = typeAsByte;
51    }
52  
53    DatumType(SerDe<Datum> serDe, int i) {
54      this(serDe, (byte) i);
55    }
56  
57    public SerDe<Datum> getSerDe() {
58      return serDe;
59    }
60  
61    public byte getTypeAsByte() {
62      return typeAsByte;
63    }
64  
65    public static DatumType fromByte(byte typeAsByte) {
66      if (typeAsByte == NULL.getTypeAsByte()) {
67        return NULL;
68      } else if (typeAsByte == BOOLEAN.getTypeAsByte()) {
69        return BOOLEAN;
70      } else if (typeAsByte == BYTE.getTypeAsByte()) {
71        return BYTE;
72      } else if (typeAsByte == CHAR.getTypeAsByte()) {
73        return CHAR;
74      } else if (typeAsByte == SHORT.getTypeAsByte()) {
75        return SHORT;
76      } else if (typeAsByte == INTEGER.getTypeAsByte()) {
77        return INTEGER;
78      } else if (typeAsByte == LONG.getTypeAsByte()) {
79        return LONG;
80      } else if (typeAsByte == FLOAT.getTypeAsByte()) {
81        return FLOAT;
82      } else if (typeAsByte == DOUBLE.getTypeAsByte()) {
83        return DOUBLE;
84      } else if (typeAsByte == STRING.getTypeAsByte()) {
85        return STRING;
86      } else if (typeAsByte == LIST.getTypeAsByte()) {
87        return LIST;
88      } else if (typeAsByte == MAP.getTypeAsByte()) {
89        return MAP;
90      } else if (typeAsByte == OTHER.getTypeAsByte()) {
91        return OTHER;
92      } else {
93        throw new IllegalArgumentException("unknown byte type: " + typeAsByte);
94      }
95    }

97    /**
98     * helper function to see if the value is a Long or less
99     *
100    * @param datum
101    * @return
102    */
103   public static boolean isLongCompatible(Datum datum) {
104     DatumType datumType = datum.getType();
105     boolean singleItemType = datumType.compareTo(BOOLEAN) >= 0 && datumType.compareTo(LONG) <= 0;
106 
107     if (!singleItemType && datumType == LIST) {
108       // only special case is List, since we promote a single-item list to a "scalar" datum
109       // recursively 
110       List<Datum> listDatum = ((ListDatum) datum).asList();
111 
112       return listDatum.size() == 1 && isLongCompatible(listDatum.get(0));
113     } else {
114       return singleItemType;
115     }
116   }
