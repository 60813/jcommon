16  package com.facebook.lifecycle;
17  
18  /**
19   * allows for objects to register shutdown hooks
20   * used instead of direct shutdown hooks in order to 
21   * manage the order they are called
22   * 
23   * The idea here is that on startup, we register the Thread this provides
24   * as a hook in case the JVM is killed.  If the fb303.shutdown() is called,
25   * this will execute and remove itself from the jvm shutdown hooks
26   */
27  public interface ShutdownManager<T extends Comparable> {
28    public boolean tryAddShutdownHook(Runnable hook);
29  
30    /**
31     * attempt to add a shutdown hook to be run at shutdown
32     * 
33     * @param stage - stage to add the hook to
34     * @param hook - hook to run
35     * @return true if the hook was added, false if we are already shutting
36     * down and the hook was not added
37     */
38    public boolean tryAddShutdownHook(T stage, Runnable hook);
39    public void addShutdownHook(Runnable hook);
40  
41    /**
42     * add a hook to a given stage 
43     *  
44     * @param stage - stage to add the hook to
45     * @param hook - hook to run
46     * 
47     * @throws IllegalArgumentException if shutdown has started and we have
48     * passed the specified stage.
49     */
50    public void addShutdownHook(T stage, Runnable hook);
51    public void shutdown();
52    public Thread getAsThread();
53  }
