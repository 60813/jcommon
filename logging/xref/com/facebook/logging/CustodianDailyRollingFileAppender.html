199 	// This method computes the roll over period by looping over the
200 	// periods, starting with the shortest, and stopping when the r0 is
201 	// different from from r1, where r0 is the epoch formatted according
202 	// the datePattern (supplied by the user) and r1 is the
203 	// epoch+nextMillis(i) formatted according to datePattern. All date
204 	// formatting is done in GMT and not local format because the test
205 	// logic is based on comparisons relative to 1970-01-01 00:00:00
206 	// GMT (the epoch).
207 
208 	int computeCheckPeriod()
209 	{
210 		RollingCalendar rollingCalendar = new RollingCalendar(gmtTimeZone,
211 				Locale.ENGLISH);
212 		// set sate to 1970-01-01 00:00:00 GMT
213 		Date epoch = new Date(0);
214 		if (datePattern != null)
215 		{
216 			for (int i = TOP_OF_MINUTE; i <= TOP_OF_MONTH; i++)
217 			{
218 				SimpleDateFormat simpleDateFormat = new SimpleDateFormat(
219 						datePattern);
220 				simpleDateFormat.setTimeZone(gmtTimeZone); // do all date
221 				// formatting in GMT
222 				String r0 = simpleDateFormat.format(epoch);
223 				rollingCalendar.setType(i);
224 				Date next = new Date(rollingCalendar.getNextCheckMillis(epoch));
225 				String r1 = simpleDateFormat.format(next);
226 				// System.out.println("Type = "+i+", r0 = "+r0+", r1 = "+r1);
227 				if (r0 != null && r1 != null && !r0.equals(r1))
228 				{
229 					return i;
230 				}
231 			}
232 		}
233 		return TOP_OF_TROUBLE; // Deliberately head for trouble...
234 	}

236 	/**
237 	 * Rollover the current file to a new file.
238 	 */
239 	void rollOver() throws IOException
240 	{
241 
242 		/* Compute filename, but only if datePattern is specified */
243 		if (datePattern == null)
244 		{
245 			errorHandler.error("Missing DatePattern option in rollOver().");
246 			return;
247 		}
248 
249 		String datedFilename = fileName + sdf.format(now);
250 		// It is too early to roll over because we are still within the
251 		// bounds of the current interval. Rollover will occur once the
252 		// next interval is reached.
253 		if (scheduledFilename.equals(datedFilename))
254 		{
255 			return;
256 		}
257 
258 		// close current file, and rename it to datedFilename
259 		this.closeFile();
260 
261 		File target = new File(scheduledFilename);
262 		if (target.exists())
263 		{
264 			target.delete();
265 		}
266 
267 		File file = new File(fileName);
268 		boolean result = file.renameTo(target);
269 		if (result)
270 		{
271 			LogLog.debug(fileName + " -> " + scheduledFilename);
272 		} else
273 		{
274 			LogLog.error("Failed to rename [" + fileName + "] to ["
275 					+ scheduledFilename + "].");
276 		}
277 
278 		try
279 		{
280 			// This will also close the file. This is OK since multiple
281 			// close operations are safe.
282 			this.setFile(fileName, false, this.bufferedIO, this.bufferSize);
283 		} catch (IOException e)
284 		{
285 			errorHandler.error("setFile(" + fileName + ", false) call failed.");
286 		}
287 		scheduledFilename = datedFilename;
288 	}

290 	/**
291 	 * This method differentiates DailyRollingFileAppender from its super class.
292 	 * 
293 	 * 
294 	 * Before actually logging, this method will check whether it is time to do
295 	 * a rollover. If it is, it will schedule the next rollover time and then
296 	 * rollover.
297 	 * */
298 	protected void subAppend(LoggingEvent event)
299 	{
300 		long n = System.currentTimeMillis();
301 		if (n >= nextCheck)
302 		{
303 			now.setTime(n);
304 			nextCheck = rc.getNextCheckMillis(now);
305 			try
306 			{
307 				cleanupAndRollOver();
308 			} 
309 			catch (IOException ioe)
310 			{
311 				LogLog.error("cleanupAndRollover() failed.", ioe);
312 			}
313 		}
314 		super.subAppend(event);
315 	}

337 	/*
338 	 * This method checks to see if we're exceeding the number of log backups
339 	 * that we are supposed to keep, and if so, deletes the offending files. It
340 	 * then delegates to the rollover method to rollover to a new file if
341 	 * required.
342 	 */
343 	protected void cleanupAndRollOver() throws IOException
344 	{
345 		// Check to see if there are already 5 files
346 		File file = new File(fileName);
347 		Calendar cal = Calendar.getInstance();
348 		int maxDays = 7;
349 		try
350 		{
351 			maxDays = Integer.parseInt(getMaxNumberOfDays());
352 		} catch (Exception e)
353 		{
354 			// just leave it at 7.
355 		}
356 		cal.add(Calendar.DATE, -maxDays);
357 		Date cutoffDate = cal.getTime();
358 		if (file.getParentFile().exists())
359 		{
360 			File[] files = file.getParentFile().listFiles(
361 					new StartsWithFileFilter(file.getName(), false));
362 			int nameLength = file.getName().length();
363 			for (int i = 0; i < files.length; i++)
364 			{
365 				String datePart = null;
366 				try
367 				{
368 					datePart = files[i].getName().substring(nameLength);
369 					Date date = sdf.parse(datePart);
370 					if (date.before(cutoffDate))
371 					{
372 						files[i].delete();
373 					}
374 					//If we're supposed to zip files and this isn't already a zip
375           else if(getCompressBackups().equalsIgnoreCase("YES") || getCompressBackups().equalsIgnoreCase("TRUE"))
376           {
377             zipAndDelete(files[i]);
378           }
379 				} 
380 				catch (Exception pe)
381 				{
382 					// This isn't a file we should touch (it isn't named
383 					// correctly)
384 				}
385 			}
386 		}
387 		rollOver();
388 	}

390 	/**
391 	   * Compresses the passed file to a .zip file, stores the .zip in the
392 	   * same directory as the passed file, and then deletes the original,
393 	   * leaving only the .zipped archive.
394 	   * @param file
395 	   */
396 	  private void zipAndDelete(File file) throws IOException
397 	  {
398 	      if(!file.getName().endsWith(".zip"))
399 	      {
400 	            File zipFile = new File(file.getParent(), file.getName() + ".zip");
401 	            FileInputStream fis = new FileInputStream(file);
402 	            FileOutputStream fos = new FileOutputStream(zipFile);
403 	            ZipOutputStream zos = new ZipOutputStream(fos);
404 	            ZipEntry zipEntry = new ZipEntry(file.getName());
405 	            zos.putNextEntry(zipEntry);
406 	            
407 	            byte[] buffer = new byte[4096];
408 	            while(true)
409 	            {
410 	                int bytesRead = fis.read(buffer);
411 	                if(bytesRead == -1) break;
412 	                else
413 	                {
414 	                    zos.write(buffer, 0, bytesRead);
415 	                }
416 	            }
417 	            zos.closeEntry();
418 	            fis.close();
419 	            zos.close();
420 	            file.delete();
421 	      }
422 	  }

426 		private String startsWith;

429 		/**

431 	     */
432 		public StartsWithFileFilter(String startsWith,

434 		{

437 			inclDirs = includeDirectories;

440 		/*

442 		 * 
443 		 * @see java.io.FileFilter#accept(java.io.File)
444 		 */
445 		public boolean accept(File pathname)
446 		{
447 			if (!inclDirs && pathname.isDirectory())
448 			{
449 				return false;
450 			} else
451 				return pathname.getName().toUpperCase().startsWith(startsWith);
452 		}
453 	}

456 /**
457  * RollingCalendar is a helper class to DailyRollingFileAppender. Given a
458  * periodicity type and the current time, it computes the start of the next
459  * interval.
460  * */
461 class RollingCalendar extends GregorianCalendar
462 {
463 	private static final long serialVersionUID = -3560331770601814177L;
464 
465 	int type = CustodianDailyRollingFileAppender.TOP_OF_TROUBLE;

470 	}

475 	}

479 		this.type = type;
480 	}

482 	public long getNextCheckMillis(Date now)
483 	{
484 		return getNextCheckDate(now).getTime();
485 	}
486 
487 	public Date getNextCheckDate(Date now)
488 	{
489 		this.setTime(now);
490 
491 		switch (type)
492 		{
493 		case CustodianDailyRollingFileAppender.TOP_OF_MINUTE:
494 			this.set(Calendar.SECOND, 0);
495 			this.set(Calendar.MILLISECOND, 0);
496 			this.add(Calendar.MINUTE, 1);
497 			break;
498 		case CustodianDailyRollingFileAppender.TOP_OF_HOUR:
499 			this.set(Calendar.MINUTE, 0);
500 			this.set(Calendar.SECOND, 0);
501 			this.set(Calendar.MILLISECOND, 0);
502 			this.add(Calendar.HOUR_OF_DAY, 1);
503 			break;
504 		case CustodianDailyRollingFileAppender.HALF_DAY:
505 			this.set(Calendar.MINUTE, 0);
506 			this.set(Calendar.SECOND, 0);
507 			this.set(Calendar.MILLISECOND, 0);
508 			int hour = get(Calendar.HOUR_OF_DAY);
509 			if (hour < 12)
510 			{
511 				this.set(Calendar.HOUR_OF_DAY, 12);
512 			} else
513 			{
514 				this.set(Calendar.HOUR_OF_DAY, 0);
515 				this.add(Calendar.DAY_OF_MONTH, 1);
516 			}
517 			break;
518 		case CustodianDailyRollingFileAppender.TOP_OF_DAY:
519 			this.set(Calendar.HOUR_OF_DAY, 0);
520 			this.set(Calendar.MINUTE, 0);
521 			this.set(Calendar.SECOND, 0);
522 			this.set(Calendar.MILLISECOND, 0);
523 			this.add(Calendar.DATE, 1);
524 			break;
525 		case CustodianDailyRollingFileAppender.TOP_OF_WEEK:
526 			this.set(Calendar.DAY_OF_WEEK, getFirstDayOfWeek());
527 			this.set(Calendar.HOUR_OF_DAY, 0);
528 			this.set(Calendar.MINUTE, 0);
529 			this.set(Calendar.SECOND, 0);
530 			this.set(Calendar.MILLISECOND, 0);
531 			this.add(Calendar.WEEK_OF_YEAR, 1);
532 			break;
533 		case CustodianDailyRollingFileAppender.TOP_OF_MONTH:
534 			this.set(Calendar.DATE, 1);
535 			this.set(Calendar.HOUR_OF_DAY, 0);
536 			this.set(Calendar.MINUTE, 0);
537 			this.set(Calendar.SECOND, 0);
538 			this.set(Calendar.MILLISECOND, 0);
539 			this.add(Calendar.MONTH, 1);
540 			break;
541 		default:
542 			throw new IllegalStateException("Unknown periodicity type.");
543 		}
544 		return getTime();
545 	}
