16  package com.facebook.stats;
17  
18  /**
19   * This was written to resemble some older libraries.  You may find
20   * more functionality in StatsUtil.java.
21   *
22   * This is the interface that mirrors
23   * fbcode/common/fb303/cpp/FacebookBase.h functionality for adding
24   * stats that track their history over 60 secs, 600 secs, 3600 secs,
25   * and "alltime".  Any class that implements HistoryMgrIf will
26   * implement FacebookServiceIf (see thrift files) as well.
27   *
28   * HistoryMgrIf manages a set of stats.  Stats are usually registered
29   * with a call to addStatExportType(name,etype) and then set or
30   * incremented many times with addStatValue(name,delta).  The export
31   * type supports reporting different summaries of the stat (sum, avg,
32   * rate, count).
33   *
34   * For example, an event counter can be exported with "sum" and "avg"
35   * and if it's incremented once a second with addStatValue("c", 10)
36   * the getCounters() call will report "c.sum.60 == 600", "c.avg.60 ==
37   * 10", "c.sum.600 = 6000", etc.  (Assuming the underlying
38   * implementation uses a MultiWindowRate counter which reports stats
39   * histories for 60, 600, 3600, and "alltime".)
40   *
41   */
42  public interface HistoryManager {
43  
44    public enum ExportType {
45      NONE(0), SUM(1), COUNT(2), AVG(4), RATE(8);
46      private int value;
47      ExportType(int v) { this.value = v; }
48      public String toString() { return super.toString().toLowerCase(); }
49      public int value() { return value; }
50    };
51  
52    /**
53     * Declares the export type of the stat.  The same name can
54     * be exported in multiple ways (e.g. avg and sum).
55     */
56    public void addStatExportType(String name, ExportType etype);
57  
58    /**
59     * Set or increment the value of the stat.
60     */
61    public void addStatValue(String name, long delta);
62  
63    /**
64     * Typically you will implement this from FacebookServiceIf
65     * to actually read the counters/stats created
66     *
67     * public Map<String, Long> getCounters();
68     */
69  
70  }
