16  package com.facebook.stats;

18  /*
19   * This is the interface that mirrors
20   * fbcode/common/fb303/cpp/FacebookBase.h functionality for adding
21   * stats that track their history over 60 secs, 600 secs, 3600 secs,
22   * and "alltime".  Any class that implements HistoryMgrIf will
23   * implement FacebookServiceIf (see thrift files) as well.
24   *
25   * HistoryMgrIf manages a set of stats.  Stats are usually registered
26   * with a call to addStatExportType(name,etype) and then set or
27   * incremented many times with addStatValue(name,delta).  The export
28   * type supports reporting different summaries of the stat (sum, avg,
29   * rate, count).
30   *
31   * For example, an event counter can be exported with "sum" and "avg"
32   * and if it's incremented once a second with addStatValue("c", 10)
33   * the getCounters() call will report "c.sum.60 == 600", "c.avg.60 ==
34   * 10", "c.sum.600 = 6000", etc.  (Assuming the underlying
35   * implementation uses a MultiWindowRate counter which reports stats
36   * histories for 60, 600, 3600, and "alltime".)
37   *
38   */
39  public interface HistoryManager {
40  
41    public enum ExportType {
42      NONE(0), SUM(1), COUNT(2), AVG(4), RATE(8);
43      private int value;
44      ExportType(int v) { this.value = v; }
45      public String toString() { return super.toString().toLowerCase(); }
46      public int value() { return value; }
47    };
48  
49    public void addStatExportType(String name, ExportType etype);
50    public void addStatValue(String name, long delta);
51  
52    /**
53     * Typically you will implement this from FacebookServiceIf
54     * to actually read the counters/stats created
55     *
56     * public Map<String, Long> getCounters();
57     */
58  
59  }
