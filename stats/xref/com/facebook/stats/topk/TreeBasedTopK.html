16  package com.facebook.stats.topk;
17  
18  import com.facebook.collections.ComparablePair;
19  import com.google.common.base.Preconditions;
20  
21  import java.util.HashMap;
22  import java.util.HashSet;
23  import java.util.LinkedList;
24  import java.util.List;
25  import java.util.Map;
26  import java.util.Set;
27  import java.util.TreeSet;
28  
29  /*
30   * Tree-based implementation of streaming top-k.
31   * This version is optimized for fast retrieval of the top k elements.
32   * Time complexity of add() is O(log k) worst case, O(1) best case.
33   * Time complexity of getTopK() is O(k).
34   * Space usage is O(n + k).
35   * (n = keySpaceSize)
36   */
37  public class TreeBasedTopK<T extends Comparable<T>> implements TopK<T> {
38    private final int k;
39    private final Map<T, Long> counts = new HashMap<T, Long>();
40    private final Set<T> topKeys;
41    private final TreeSet<ComparablePair<Long, T>> topPairs = new TreeSet<ComparablePair<Long, T>>();
42    private long smallestTopCount = Long.MAX_VALUE;
43  
44    public TreeBasedTopK(int k) {
45      this.k = k;
46      topKeys = new HashSet<T>(k);
47    }
48  
49    /*
50     * Cost of an add() operation is O(log k) if the candidate has to be inserted in
51     * the current top-k elements, O(1) otherwise.
52     */
53    @Override
54    public synchronized void add(T key, long count) {
55      Preconditions.checkNotNull(key, "key can't be null");
56      Preconditions.checkArgument(count >= 0, "count to add must be non-negative, got %s", count);
57  
58      if (count == 0) {
59        return;
60      }
61  
62      Long currentCount = counts.get(key);
63  
64      if (currentCount == null) {
65        currentCount = 0L;
66      }
67  
68      long updatedCount = currentCount + count;
69  
70      counts.put(key, updatedCount);
71  
72      if (topKeys.contains(key)) {
73        topPairs.remove(new ComparablePair<Long, T>(currentCount, key));
74        topPairs.add(new ComparablePair<Long, T>(updatedCount, key));
75      } else if (topPairs.size() < k) {
76        topPairs.add(new ComparablePair<Long, T>(updatedCount, key));
77        topKeys.add(key);
78        smallestTopCount = Math.min(smallestTopCount, updatedCount);
79      } else if (updatedCount > smallestTopCount) {
80        ComparablePair<Long, T> smallestTopPair = topPairs.pollFirst();
81  
82        topKeys.remove(smallestTopPair.getSecond());
83        topPairs.add(new ComparablePair<Long, T>(updatedCount, key));
84        topKeys.add(key);
85        smallestTopCount = topPairs.first().getFirst();
86      }
87    }
88  
89    @Override
90    public synchronized List<T> getTopK() {
91      LinkedList<T> topK = new LinkedList<T>();
92  
93      for (ComparablePair<Long, T> pair : topPairs) {
94        topK.addFirst(pair.getSecond());
95      }
96  
97      return topK;
98    }
99  }
