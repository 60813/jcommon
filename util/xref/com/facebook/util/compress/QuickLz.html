46    final private static int HASH_VALUES = 4096;
47    final private static int MINOFFSET = 2;
48    final private static int UNCONDITIONAL_MATCHLEN = 6;
49    final private static int UNCOMPRESSED_END = 4;
50    final private static int CWORD_LEN = 4;
51    final private static int DEFAULT_HEADERLEN = 9;

53    static int headerLen(byte[] source)
54    {
55      return ((source[0] & 2) == 2) ? 9 : 3;
56    }
57  
58    static public long sizeDecompressed(byte[] source)
59    {
60      if (headerLen(source) == 9)
61        return fastread(source, 5, 4);
62      else
63        return fastread(source, 2, 1);
64    }

66    static public long sizeCompressed(byte[] source)
67    {
68      if (headerLen(source) == 9)
69        return fastread(source, 1, 4);
70      else
71        return fastread(source, 1, 1);
72    }

74    public static byte[] compress(byte[] source)
75    {
76      int src = 0;
77      int headerlen = DEFAULT_HEADERLEN;
78      int dst = headerlen + CWORD_LEN;
79      long cword_val = 0x80000000L;
80      int cword_ptr = headerlen;
81      byte[] destination = new byte[source.length + 400];
82      int[] hashtable = new int[HASH_VALUES];
83      int[] cachetable = new int[HASH_VALUES];
84      byte[] hash_counter = new byte[HASH_VALUES];
85      byte[] d2;
86      int fetch = 0;
87      int last_matchstart = (source.length - UNCONDITIONAL_MATCHLEN - 
88          UNCOMPRESSED_END - 1);
89  
90      if(source.length == 0)
91        return new byte[0];
92  
93      if(src <= last_matchstart)
94        fetch = (int)fastread(source, src, 3);
95  
96      while (src <= last_matchstart)
97      {
98        if ((cword_val & 1) == 1)
99        {
100         if (src > 3 * (source.length >> 2) && dst > src - (src >> 5))
101         {
102           d2 = new byte[source.length + DEFAULT_HEADERLEN];
103           d2[0] = 2 | 0;
104           fastwrite(d2, 1, source.length + headerlen, 4);
105           fastwrite(d2, 5, source.length, 4);
106           System.arraycopy(source, 0, d2, headerlen, source.length);
107           return d2;
108         }
109 
110         fastwrite(destination, cword_ptr, (cword_val >>> 1) | 0x80000000L, 4);
111         cword_ptr = dst;
112         dst += CWORD_LEN;
113         cword_val = 0x80000000L;
114       }            
115 
116       int hash = ((fetch >>> 12) ^ fetch) & (HASH_VALUES - 1);
117       int o = hashtable[hash];
118       int cache = cachetable[hash] ^ fetch;
119 
120       cachetable[hash] = fetch;
121       hashtable[hash] = src;
122 
123       if (cache == 0 && src - o > MINOFFSET && hash_counter[hash] != 0)
124       {
125         cword_val = ((cword_val >>> 1) | 0x80000000L);
126         if (source[o + 3] != source[src + 3])
127         {
128           int f = 3 - 2 | (hash << 4);
129           destination[dst + 0] = (byte)(f >>> 0 * 8);
130           destination[dst + 1] = (byte)(f >>> 1 * 8);
131           src += 3;
132           dst += 2;
133         }
134         else
135         {
136           int old_src = src;
137           int remaining = ((source.length - UNCOMPRESSED_END - src + 1 - 1)>255 
138               ? 255 
139                   : (source.length - UNCOMPRESSED_END - src + 1 - 1));
140 
141           src += 4;
142           if (source[o + src - old_src] == source[src])
143           {
144             src++;
145             if (source[o + src - old_src] == source[src])
146             {
147               src++;
148               while (source[o + (src - old_src)] == source[src] && 
149                   (src - old_src) < remaining)
150                 src++;
151             }
152           }
153 
154           int matchlen = src - old_src;
155 
156           hash <<= 4;
157           if (matchlen < 18)
158           {
159             int f = hash | (matchlen - 2);
160             // Neither Java nor C# wants to inline fastwriteN
161             destination[dst + 0] = (byte)(f >>> 0 * 8);
162             destination[dst + 1] = (byte)(f >>> 1 * 8);
163             dst += 2;
164           }
165           else
166           {
167             int f = hash | (matchlen << 16);
168             fastwrite(destination, dst, f, 3);
169             dst += 3;
170           }
171         }
172         fetch = (int)fastread(source, src, 3);
173       }
174       else
175       {
176         hash_counter[hash] = 1;
177         destination[dst] = source[src];
178         cword_val = (cword_val >>> 1);
179         src++;
180         dst++;
181         fetch = ((fetch >>> 8) & 0xffff) | 
182         ((((int)source[src + 2]) & 0xff) << 16);
183       }
184     }
185 
186     while (src <= source.length - 1)
187     {
188       if ((cword_val & 1) == 1)
189       {
190         fastwrite(destination, cword_ptr, 
191             (long)((cword_val >>> 1) | 0x80000000L), 4);
192         cword_ptr = dst;
193         dst += CWORD_LEN;
194         cword_val = 0x80000000L;
195       }
196 
197       destination[dst] = source[src];
198       src++;
199       dst++;
200       cword_val = (cword_val >>> 1);
201     }
202     while ((cword_val & 1) != 1)
203     {
204       cword_val = (cword_val >>> 1);
205     }
206     fastwrite(destination, cword_ptr, (long)((cword_val >>> 1) | 0x80000000L), 
207         CWORD_LEN);
208     destination[0] = 2 | 1;
209     fastwrite(destination, 1, (long)dst, 4);
210     fastwrite(destination, 5, (long)source.length, 4);
211     d2 = new byte[dst];
212     System.arraycopy(destination, 0, d2, 0, dst);      
213     return d2;
214   }

216   static long fastread(byte[] a, int i, int numbytes)
217   {
218     long l = 0;
219     switch (numbytes)
220     {
221     case 3:
222       l |= ((((int)a[i + 0]) & 0xffL) << 0*8);
223       l |= ((((int)a[i + 1]) & 0xffL) << 1*8);
224       l |= ((((int)a[i + 2]) & 0xffL) << 2*8);
225       break;
226 
227     case 2:
228       l |= ((((int)a[i + 0]) & 0xffL) << 0*8);
229       l |= ((((int)a[i + 1]) & 0xffL) << 1*8);
230       break;
231     case 1:
232       l |= ((((int)a[i + 0]) & 0xffL) << 0*8);
233       break;
234     case 4:
235       l |= ((((int)a[i + 0]) & 0xffL) << 0*8);
236       l |= ((((int)a[i + 1]) & 0xffL) << 1*8);
237       l |= ((((int)a[i + 2]) & 0xffL) << 2*8);
238       l |= ((((int)a[i + 3]) & 0xffL) << 3*8);
239       break;
240     }
241     return l;
242   }

244   static void fastwrite(byte[] a, int i, long value, int numbytes)
245   {
246     switch (numbytes)
247     {
248     case 3:
249       a[i] = (byte)value;
250       a[i + 1] = (byte)(value >>> 8);
251       a[i + 2] = (byte)(value >>> 16);
252       break;
253     case 2:
254       a[i] = (byte)value;
255       a[i + 1] = (byte)(value >>> 8);
256       break;
257     case 4:
258       a[i] = (byte)value;
259       a[i + 1] = (byte)(value >>> 8);
260       a[i + 2] = (byte)(value >>> 16);
261       a[i + 3] = (byte)(value >>> 24);
262       break;
263     }
264   }

266   static public byte[] decompress(byte[] source)
267   {
268     int size = (int)sizeDecompressed(source);
269 
270     int src = headerLen(source);
271     int dst = 0;
272     long cword_val = 1;
273     byte[] destination = new byte[size];
274     int[] hashtable = new int[4096];
275     byte[] hash_counter = new byte[4096];
276     int last_matchstart = size - UNCONDITIONAL_MATCHLEN - UNCOMPRESSED_END - 1;
277     int last_hashed = -1;
278     int hash;
279     int fetch = 0;
280 
281     if ((source[0] & 1) != 1)
282     {
283       byte[] d2 = new byte[size];
284       System.arraycopy(source, headerLen(source), d2, 0, size);
285       return d2;
286     }
287 
288     for (; ; )
289     {
290       if (cword_val == 1)
291       {
292         cword_val = fastread(source, src, 4);
293         src += 4;
294         if (dst <= last_matchstart)
295           fetch = (int)fastread(source, src, 3);
296       }
297 
298       if ((cword_val & 1) == 1)
299       {
300         int matchlen;
301         int offset2;
302 
303         cword_val = cword_val >>> 1;
304       hash = (fetch >>> 4) & 0xfff;
305       offset2 = hashtable[hash];
306 
307       if ((fetch & 0xf) != 0)
308       {
309         matchlen = (fetch & 0xf) + 2;
310         src += 2;
311       }
312       else
313       {
314         matchlen = ((int)source[src + 2]) & 0xff;
315         src += 3;
316       }
317 
318       destination[dst + 0] = destination[offset2 + 0];
319       destination[dst + 1] = destination[offset2 + 1];
320       destination[dst + 2] = destination[offset2 + 2];
321 
322       for (int i = 3; i < matchlen; i += 1)
323       {
324         destination[dst + i] = destination[offset2 + i];
325       }
326       dst += matchlen;
327 
328       fetch = (int)fastread(destination, last_hashed + 1, 3); 
329       // destination[last_hashed + 1] | (destination[last_hashed + 2] << 8) | 
330       // (destination[last_hashed + 3] << 16);
331       while (last_hashed < dst - matchlen)
332       {
333         last_hashed++;
334         hash = ((fetch >>> 12) ^ fetch) & (HASH_VALUES - 1);
335         hashtable[hash] = last_hashed;
336         hash_counter[hash] = 1;
337         fetch = fetch >>> 8 & 0xffff | 
338         (((int)destination[last_hashed + 3]) & 0xff) << 16;
339       }
340       last_hashed = dst - 1;
341       fetch = (int)fastread(source, src, 3);
342       }
343       else
344       {
345         if (dst <= last_matchstart)
346         {
347           destination[dst] = source[src];
348           dst += 1;
349           src += 1;
350           cword_val = cword_val >>> 1;
351 
352         while (last_hashed < dst - 3)
353         {
354           last_hashed++;
355           int fetch2 = (int)fastread(destination, last_hashed, 3);
356           hash = ((fetch2 >>> 12) ^ fetch2) & (HASH_VALUES - 1);
357           hashtable[hash] = last_hashed;
358           hash_counter[hash] = 1;
359         }
360         fetch = fetch >> 8 & 0xffff | (((int)source[src + 2]) & 0xff) << 16;  
361         }
362         else
363         {
364           while (dst <= size - 1)
365           {
366             if (cword_val == 1)
367             {
368               src += CWORD_LEN;
369               cword_val = 0x80000000L;
370             }
371 
372             destination[dst] = source[src];
373             dst++;
374             src++;
375             cword_val = cword_val >>> 1;
376           }
377 
378           byte[] d2 = new byte[size];
379           System.arraycopy(destination, 0, d2, 0, size);
380           return d2;
381         }
382       }
383     }
384   }
