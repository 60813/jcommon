16  package com.facebook.zookeeper;
17  
18  import java.nio.charset.Charset;
19  import java.util.ArrayList;
20  import java.util.List;
21  
22  public class ZkUtil {
23    private static final Charset CHARSET = Charset.forName("UTF-8");
24  
25    /**
26     * Don't allow construction of this class
27     */
28    private ZkUtil() {
29    }
30  
31    /**
32     * Converts String into a byte array using a fixed character set. All
33     * classes making this conversion should use this function to ensure a
34     * consistent data representation in ZooKeeper.
35     * @param str
36     * @return byte array representation of String
37     */
38    public static byte[] stringToBytes(String str) {
39      return str.getBytes(CHARSET);
40    }
41  
42    /**
43     * Converts a byte array from ZooKeeper into a String.
44     * @param data
45     * @return String representation of byte array
46     */
47    public static String bytesToString(byte[] data) {
48      return data == null ? null : new String(data, CHARSET);
49    }
50  
51    /**
52     * Filters the given node list by the given prefixes.
53     * This method is all-inclusive--if any element in the node list starts
54     * with any of the given prefixes, then it is included in the result.
55     *
56     * @param nodes the nodes to filter
57     * @param prefixes the prefixes to include in the result
58     * @return list of every element that starts with one of the prefixes
59     */
60    public static List<String> filterByPrefix(
61      List<String> nodes, String... prefixes
62    ) {
63      List<String> lockChildren = new ArrayList<String>();
64      for (String child : nodes){
65        for (String prefix : prefixes){
66          if (child.startsWith(prefix)){
67            lockChildren.add(child);
68            break;
69          }
70        }
71      }
72      return lockChildren;
73    }
74  }
