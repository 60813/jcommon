34  /**
35   * ZkApplication is an abstract base class that provides a template-and-hook
36   * style ZooKeeper state management framework.
37   *
38   * Features:
39   * - Automatic connection and reconnection to ZooKeeper on disconnects and
40   *   expirations.
41   * - Issues initialize(), repair(), and expire() callbacks to the application
42   *   as various connection events occur.
43   * - Models application state with a finite state machine. States are queryable
44   *   from subclasses.
45   *
46   * =============================== STATES ===============================
47   *
48   * PRESTART:
49   * This is the application's initial state before the user issues a start()
50   * command to begin the application
51   *
52   * DISCONNECTED:
53   * Application is not connected to ZooKeeper and does not have any
54   * application state set in ZooKeeper (e.g. watches or ephemeral nodes).
55   * Entry to this state automatically triggers a connection loop that retries
56   * until successful.
57   *
58   * CONNECTED:
59   * Application is connected to ZooKeeper, but application state in ZooKeeper
60   * has not been fully initialized. Entry to this state automatically triggers
61   * callbacks to the initialize() abstract method, looping until initialize()
62   * returns true.
63   *
64   * FUNCTIONAL:
65   * Application is connected to ZooKeeper and has full application state
66   * initialized in ZooKeeper. At this point the application should be fully
67   * functioning.
68   *
69   * SAFEMODE:
70   * Application was successfully initialized, but became disconnected. Since
71   * many applications will cache ZooKeeper state, this state signifies a
72   * cache read-only mode where ZooKeeper is unavailable. Entry to this
73   * state will automatically trigger a connection loop that retries until
74   * reconnected, or until the session expires.
75   *
76   * SAFEMODE_REPAIR:
77   * Application in safemode was successfully reconnected to ZooKeeper without
78   * expiration. Entry to this state will automatically trigger callbacks to the
79   * repair() abstract method, looping until repair() returns true. While
80   * initialize() and repair() may do the same things, we make this distinction
81   * as it is often possible to optimize the repair method to only repeat failed
82   * commands.
83   *
84   * SHUTDOWN:
85   * Application has been shut down by the user via the shutdown() method.
86   *
87   * ======================= STATE TRANSITION DIAGRAM =====================
88   *
89   * The state transition diagram appears as follows:
90   *
91   *                  PRESTART     (expire*)
92   *                     | (start)    |
93   *                     v            |      
94   *                DISCONNECTED <----       -->  SAFEMODE
95   *                     | (connect)        /        | (connect)
96   *                     v                 /         v
97   *                 CONNECTED    (dc'ed) /    SAFEMODE_REPAIR
98   *                       \             /          /
99   *                        \           /          /
100  *                 (init)  \         /          / (repair)
101  *                          \       /          /
102  *                           v     /          /
103  *                          FUNCTIONAL <-----
104  *
105  *
106  *                  (shutdown*) -----> SHUTDOWN
107  *
108  * NOTE: '*' denotes an event that unconditionally leads to a specific state,
109  * regardles of the pre-existing state.
110  */
111 public abstract class ZkApplication {
112   private static final Logger APP_LOG = Logger.getLogger(ZkApplication.class);

115   private final ExecutorService watchExecutor;

117   private final long retryIntervalMillis;
118   private final CountDownLatch initLatch = new CountDownLatch(1);

120   private volatile boolean isStarted = false;

164   // Instance must be started before it becomes valid for use
165   public synchronized void start() {
166     if (isStarted) {
167       throw new IllegalStateException("Should only be started once");
168     }
169     ZooKeeper.States zkState =
170       zkConnectionManager.registerWatcher(new ConnectionWatcher());
171     // Synchronize our application state with ZooKeeper state
172     context.start((zkState == ZooKeeper.States.CONNECTED)
173       ? State.CONNECTED
174       : State.DISCONNECTED
175     );
176     isStarted = true;
177     // Allow watch signals to pass only after initialization
178     initLatch.countDown();
179   }
180 
181   public boolean isFunctional() {
182     return context.getState() == State.FUNCTIONAL;
183   }
184 
185   public boolean isSafeMode() {
186     return context.getState() == State.SAFEMODE ||
187       context.getState() == State.SAFEMODE_REPAIR;
188   }
189 
190   public boolean isShutdown() {
191     return context.getState() == State.SHUTDOWN;
192   }
193 
194   public synchronized void shutdown() {
195     if (!isStarted) {
196       throw new IllegalStateException("Application not yet started");
197     }
198     context.shutdown();
199     watchExecutor.shutdown();
200     retryExecutor.shutdown();
201   }

205   /**
206    * Initializes the application such that it is fully functioning.
207    * Implementations should be idempotent, and will be retried repeatedly
208    * until it succeeds.
209    * @return true if successfully initialized, false otherwise
210    */
211   protected abstract boolean initialize();
212 
213   /**
214    * Repair the application state to a fully functioning state following
215    * a ZooKeeper disconnect. Implementations should be idempotent, and will
216    * be retried repeatedly until it succeeds.
217    * @return true if successfully repaired, false otherwise
218    */
219   protected abstract boolean repair();
220 
221   /**
222    * Cleans up application state as necessary following a ZooKeeper session
223    * expiration. Implementations should be idempotent.
224    */
225   protected abstract void expire();

230   private class ConnectionWatcher implements Watcher {
231     @Override
232     public void process(final WatchedEvent event) {
233       watchExecutor.execute(new ErrorLoggingRunnable(new Runnable() {
234         @Override
235         public void run() {
236           try {
237             initLatch.await(); // Wait until we have been fully initialized
238           } catch (InterruptedException e) {
239             APP_LOG.error("Init latch interrupted, continuing...");
240             Thread.currentThread().interrupt();
241           }
242           context.handleEvent(event.getState());
243         }
244       }));
245     }
246   }

248   private interface StateHandler {
249     void handleEvent(Watcher.Event.KeeperState event);
250     void inboundHook();
251     void outboundHook();
252   }
253 
254   private class StateContext {
255     private volatile State state = State.PRESTART;
256     private final Object transitionLock = new Object();
257     private final Map<State, StateHandler> handlerCache =
258       new EnumMap<State, StateHandler>(State.class);
259 
260     private StateContext() {
261       // Should be one entry per possible state
262       handlerCache.put(State.PRESTART, new PreStartStateHandler());
263       handlerCache.put(State.DISCONNECTED, new DisconnectedStateHandler());
264       handlerCache.put(State.CONNECTED, new ConnectedStateHandler());
265       handlerCache.put(State.FUNCTIONAL, new FunctionalStateHandler());
266       handlerCache.put(State.SAFEMODE, new SafeModeStateHandler());
267       handlerCache.put(State.SAFEMODE_REPAIR, new SafeModeRepairStateHandler());
268       handlerCache.put(State.SHUTDOWN, new ShutdownStateHandler());
269     }
270 
271     // Instance must be started before it becomes valid for use
272     public void start(State initialState) {
273       synchronized (transitionLock) {
274         transition(initialState);
275       }
276     }
277 
278     public void handleEvent(Watcher.Event.KeeperState event) {
279       synchronized (transitionLock) {
280         getHandler().handleEvent(event);
281       }
282     }
283 
284     public void shutdown() {
285       synchronized (transitionLock) {
286         if (state == State.SHUTDOWN) {
287           APP_LOG.warn("Multiple shutdown calls");
288           return;
289         }
290         internalTransition(State.SHUTDOWN);
291       }
292     }
293 
294     public State getState() {
295       return state;
296     }
297 
298     private StateHandler getHandler() {
299       return handlerCache.get(state);
300     }
301 
302     private void transition(State newState) {
303       if (newState == State.SHUTDOWN) {
304         throw new IllegalArgumentException(
305           "Set SHUTDOWN state by calling the shutdown method"
306         );
307       }
308       // Shutdown is a terminal state
309       if (state != State.SHUTDOWN) {
310         internalTransition(newState);
311       }
312     }
313 
314     private void internalTransition(State newState) {
315       getHandler().outboundHook();
316       state = newState;
317       getHandler().inboundHook();
318     }

360     private class ConnectedStateHandler implements StateHandler {
361       private volatile boolean isActive = false;
362       private volatile boolean isScheduled = false;
363       private final Object scheduleCheckLock = new Object();
364 
365       @Override
366       public void handleEvent(Watcher.Event.KeeperState event) {
367         switch (event) {
368           case SyncConnected:
369             break;
370           case Disconnected:
371             context.transition(State.DISCONNECTED);
372             break;
373           case Expired:
374             context.transition(State.DISCONNECTED);
375             expire();
376             break;
377         }
378       }
379 
380       @Override
381       public void inboundHook() {
382         startInitLoop();
383       }
384 
385       private void startInitLoop() {
386         isActive = true;
387         if (scheduleCompareAndSet()) {
388           retryExecutor.execute(new ErrorLoggingRunnable(new Runnable() {
389             @Override
390             public void run() {
391               try {
392                 synchronized (transitionLock) {
393                   // Only initialize if we are in the same state
394                   if (context.getState() == State.CONNECTED) {
395                     // Attempt initialization
396                     if (initialize()) {
397                       context.transition(State.FUNCTIONAL);
398                       return; // Success, don't reschedule
399                     }
400                   }
401                 }
402               } finally {
403                 isScheduled = false;
404               }
405               if (scheduleCompareAndSet()) {
406                 retryExecutor.schedule(
407                   this, retryIntervalMillis, TimeUnit.MILLISECONDS
408                 );
409               }
410             }
411           }));
412         }
413       }
414 
415       private boolean scheduleCompareAndSet() {
416         synchronized (scheduleCheckLock) {
417           if (isActive && !isScheduled) {
418             isScheduled = true;
419             return true;
420           }
421           return false;
422         }
423       }
424 
425       @Override
426       public void outboundHook() {
427         stopRepairLoop();
428       }
429 
430       private void stopRepairLoop() {
431         isActive = false;
432       }
433     }

484     private class SafeModeRepairStateHandler implements StateHandler {
485       private volatile boolean isActive = false;
486       private volatile boolean isScheduled = false;
487       private final Object scheduleCheckLock = new Object();
488 
489       public void handleEvent(Watcher.Event.KeeperState event) {
490         switch (event) {
491           case SyncConnected:
492             break;
493           case Disconnected:
494             context.transition(State.SAFEMODE);
495             break;
496           case Expired:
497             context.transition(State.DISCONNECTED);
498             expire();
499             break;
500         }
501       }
502 
503       @Override
504       public void inboundHook() {
505         startRepairLoop();
506       }
507 
508       private void startRepairLoop() {
509         isActive = true;
510         if (scheduleCompareAndSet()) {
511           retryExecutor.execute(new ErrorLoggingRunnable(new Runnable() {
512             @Override
513             public void run() {
514               try {
515                 synchronized (transitionLock) {
516                   // Only repair if we are still in the same state
517                   if (context.getState() == State.SAFEMODE_REPAIR) {
518                     // Attempt repair
519                     if (repair()) {
520                       context.transition(State.FUNCTIONAL);
521                       return;  // Success, don't reschedule
522                     }
523                   }
524                 }
525               } finally {
526                 isScheduled = false;
527               }
528               if (scheduleCompareAndSet()) {
529                 retryExecutor.schedule(
530                   this, retryIntervalMillis, TimeUnit.MILLISECONDS
531                 );
532               }
533             }
534           }));
535         }
536       }
537 
538       private boolean scheduleCompareAndSet() {
539         synchronized (scheduleCheckLock) {
540           if (isActive && !isScheduled) {
541             isScheduled = true;
542             return true;
543           }
544           return false;
545         }
546       }
547 
548       @Override
549       public void outboundHook() {
550         stopRepairLoop();
551       }

555       }
