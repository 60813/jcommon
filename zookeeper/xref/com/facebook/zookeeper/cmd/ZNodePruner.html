55    public void prunePersistent(String pathStr, String keyword, File saveTemplate)
56      throws IOException, InterruptedException, KeeperException {
57      InputStream in = new FileInputStream(saveTemplate);
58      try {
59        prunePersistent(pathStr, keyword, in);
60      } finally {
61        in.close();
62      }
63    }
64  
65    public void prunePersistent(String pathStr, String keyword, InputStream in)
66      throws IOException, InterruptedException, KeeperException {
67      prunePersistent(pathStr, keyword, StreamImporter.importLines(in));
68    }

70    public void prunePersistent(
71      String pathStr, String keyword, List<String> toKeep
72    ) throws InterruptedException, KeeperException {
73      ZkGenericPath path = new ZkGenericPath(pathStr);
74      Set<String> keepSet = expandToKeepSet(toKeep);
75      internalPrunePersistent(path, keyword, keepSet);
76    }
77  
78    private void internalPrunePersistent(
79      ZkGenericPath path, String keyword, Set<String> keepSet
80    ) throws InterruptedException, KeeperException {
81      if (isEphemeral(path.toString())) {
82        // We should only be scanning persistent nodes (although we may delete
83        // ephemeral nodes in order to remove a parent persistent node)
84        return;
85      }
86      if (keepSet.contains(path.toString())) {
87        try {
88          ZooKeeperIface zk = getZk();
89  
90          byte[] data = zk.getData(path.toString(), false, null);
91  
92          if (data != null &&
93            keyword.equals(ZkUtil.bytesToString(data))
94            ) {
95            zk.setData(path.toString(), new byte[0], -1);
96          }
97  
98          List<String> children = zk.getChildren(path.toString(), null);
99          for (String child : children) {
100           internalPrunePersistent(path.appendChild(child), keyword, keepSet);
101         }
102       } catch (KeeperException.NoNodeException e) {
103         // If the node disappears while scanning it, then just ignore
104       }
105     } else {
106       deleteSubtree(path, keyword);
107     }
108   }
109 
110   private void deleteSubtree(ZkGenericPath path, String keyword)
111     throws InterruptedException, KeeperException {
112     ZooKeeperIface zk = getZk();
113     try {
114       if (!isEphemeral(path.toString())) {
115         // Only set freeze on non-ephemerals since ephemerals cant have children
116         zk.setData(
117           path.toString(),
118           ZkUtil.stringToBytes(keyword),
119           -1
120         );
121       }
122 
123       while (true) {
124         List<String> children = zk.getChildren(path.toString(), null);
125         for (String child : children) {
126           deleteSubtree(path.appendChild(child), keyword);
127         }
128         try {
129           zk.delete(path.toString(), -1);
130           break;
131         } catch (KeeperException.NotEmptyException e) {
132           // Repeat since children were re-added
133         }
134       }
135     } catch (KeeperException.NoNodeException e) {
136       // Ignore since we are trying to delete it
137     }
138     if (verbose) {
139       System.out.println("Deleted ZNode: " + path);
140     }
141   }

143   private boolean isEphemeral(String pathStr)
144     throws InterruptedException, KeeperException {
145     ZooKeeperIface zk = getZk();
146     Stat stat = new Stat();
147     zk.getData(pathStr, null, stat);
148     return stat.getEphemeralOwner() != 0;
149   }
150 
151   private Set<String> expandToKeepSet(List<String> toKeep) {
152     // Add all listed ZNodes as well as all of their ancestors
153     Set<String> keepSet = new HashSet<String>();
154     for (String pathStr : toKeep) {
155       ZkGenericPath path = ZkGenericPath.parse("/", pathStr);
156       Iterator<ZkGenericPath> lineageIter = path.lineageIterator();
157       while(lineageIter.hasNext()) {
158         keepSet.add(lineageIter.next().toString());
159       }
160     }
161     return keepSet;
162   }

169   @Override
170   protected Options getSpecificOptions() {
171     Options options = new Options();
172     options.addOption(
173       "z",
174       "zkpath",
175       true,
176       "ZooKeeper path to prune (includes all descendants) [Required]"
177     );
178     options.addOption(
179       "k",
180       "keyword",
181       true,
182       "Keyword to write into zNodes that will be removed. This may be used " +
183         "to help reduce activity on or underneath this node [Default: freeze]"
184     );
185     options.addOption(
186       "f",
187       "file-template",
188       true,
189       "Path to file containing new-line delimited list of ZNodes to save. " +
190         "The list only applies to persistent ZNodes. If this parameter is " +
191         "not specified, expects the values to be provided via standard input."
192     );
193     options.addOption(
194       "v",
195       "verbose",
196       false,
197       "Print verbose messages [Default: off]"
198     );
199     return options;
200   }

202   @Override
203   protected boolean verifySpecificOptions(CommandLine cmd) {
204     setVerbose(cmd.hasOption("verbose"));
205     if (!cmd.hasOption("zkpath")) {
206       System.err.println("Error: You must specify a ZooKeeper path.\n");
207       return false;
208     }
209     if (cmd.hasOption("file-template")) {
210       File template = new File(cmd.getOptionValue("file-template"));
211       if (!template.exists()) {
212         System.err.println("Error: invalid file-template path.\n");
213         return false;
214       }
215     }
216     return true;
217   }
218 
219   @Override
220   protected void runScript(CommandLine cmd) throws Exception {
221     String root = cmd.getOptionValue("zkpath");
222     String keyword = cmd.getOptionValue("keyword", ZNodePruner.DEFAULT_KEYWORD);
223     if (cmd.hasOption("file-template")) {
224       File saveTemplate = new File(cmd.getOptionValue("file-template"));
225       prunePersistent(root, keyword, saveTemplate);
226     } else {
227       prunePersistent(root, keyword, System.in);
228     }
229   }

231   public static void main(String[] args) throws Exception {
