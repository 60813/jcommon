16  package com.facebook.zookeeper.cmd;
17  
18  import com.facebook.util.StreamImporter;
19  import com.facebook.zookeeper.convenience.ZkQuickConnectionManager;
20  import com.facebook.zookeeper.convenience.ZkScript;
21  import org.apache.commons.cli.CommandLine;
22  import org.apache.commons.cli.Options;
23  
24  import java.io.File;
25  import java.io.FileInputStream;
26  import java.io.InputStream;
27  import java.util.List;
28  
29  public class ZNodeSync extends ZkScript {
30    private final ZNodeBulkLoader loader;
31    private final ZNodePruner pruner;
32  
33    public ZNodeSync(ZkQuickConnectionManager zkQuickConnectionManager) {
34      super(zkQuickConnectionManager);
35      loader = new ZNodeBulkLoader(zkQuickConnectionManager);
36      pruner = new ZNodePruner(zkQuickConnectionManager);
37      // Share the provided connection
38    }
39  
40    public ZNodeSync() {
41      this(new ZkQuickConnectionManager());
42    }
43  
44    public void setVerbose(boolean verbose) {
45      loader.setVerbose(verbose);
46      pruner.setVerbose(verbose);
47    }
48  
49    @Override
50    protected String getName() {
51      return ZNodeSync.class.getName();
52    }
53  
54    @Override
55    protected Options getSpecificOptions() {
56      Options options = new Options();
57      options.addOption(
58        "z",
59        "zkpath",
60        true,
61        "ZooKeeper path to sync (includes all descendants) [Required]"
62      );
63      options.addOption(
64        "k",
65        "keyword",
66        true,
67        "Keyword to write into zNodes that will be removed. This may be used " +
68          "to help reduce activity on or underneath this node [Default: freeze]"
69      );
70      options.addOption(
71        "f",
72        "file-template",
73        true,
74        "Path to file containing new-line delimited list of ZNodes to sync. " +
75          "The list only applies to persistent ZNodes. If this parameter is " +
76          "not specified, expects the values to be provided via standard input."
77      );
78      options.addOption(
79        "v",
80        "verbose",
81        false,
82        "Print verbose messages [Default: off]"
83      );
84      return options;
85    }
86  
87    @Override
88    protected boolean verifySpecificOptions(CommandLine cmd) {
89      if (!cmd.hasOption("zkpath")) {
90        System.err.println("Error: You must specify a ZooKeeper path.\n");
91        return false;
92      }
93      if (cmd.hasOption("file-template")) {
94        File template = new File(cmd.getOptionValue("file-template"));
95        if (!template.exists()) {
96          System.err.println("Error: invalid file-template path.\n");
97          return false;
98        }
99      }
100     return true;
101   }
102 
103   @Override
104   protected void runScript(CommandLine cmd) throws Exception {
105     setVerbose(cmd.hasOption("verbose"));
106     String root = cmd.getOptionValue("zkpath");
107     String keyword = cmd.getOptionValue("keyword", ZNodePruner.DEFAULT_KEYWORD);
108     InputStream in =
109       (cmd.hasOption("file-template"))
110         ? new FileInputStream(new File(cmd.getOptionValue("file-template")))
111         : System.in;
112     try {
113       List<String> template = StreamImporter.importLines(in);
114       // First run the bulk loader
115       loader.load(template);
116       // Then prune everything else under specified root
117       pruner.prunePersistent(root, keyword, template);
118     } finally {
119       in.close();
120     }
121   }
122 
123   public static void main(String[] args) throws Exception {
124     ZkScript script = new ZNodeSync();
125     script.runMain(args);
126     System.out.println("DONE");
127   }
